GUNLUK_HAYAT_ALGORITMASI

Mukemmel_Omlet_Yapma_Algorıtması
BAŞLA
    // 1. Malzemeleri Hazırla
    INPUT yumurta_sayisi
    INPUT tuz_miktari
    INPUT süt_miktari
    INPUT yağ_türü
    INPUT yağ_miktari
    INPUT isteğe_bağlı_malzemeler (peynir, mantar, biber, vs.)

    // 2. Malzemeleri Kontrol Et
    IF yumurta_sayisi < 2 THEN
        PRINT "En az 2 yumurta kullanmalısınız."
        EXIT
    ENDIF

    // 3. Yumurtaları Kır ve Karıştır
    FOR i FROM 1 TO yumurta_sayisi DO
        KIR yumurta[i] INTO karıştırma_kabı
    ENDFOR

    EKLE tuz_miktari TO karıştırma_kabı
    EKLE süt_miktari TO karıştırma_kabı

    // 4. İsteğe Bağlı Malzemeleri Ekle
    FOR EACH malzeme IN isteğe_bağlı_malzemeler DO
        DOĞRA malzeme
        EKLE malzeme TO karıştırma_kabı
    ENDFOR

    // 5. Karışımı Çırp
    WHILE karışım_homojen DEĞİL DO
        ÇIRP karıştırma_kabı
    ENDWHILE

    // 6. Tavayı Hazırla
    SEÇ tavayı (yapışmaz_tava)
    EKLE yağ_türü VE yağ_miktari TO tava
    ISIT tava ORTA_ATEŞTE

    // 7. Karışımı Tavaya Dök
    DÖK karışım TO tava

    // 8. Pişirme Süreci
    WHILE omlet_altı_pişmedi DO
        BEKLE 30 saniye
        KONTROL et alt_kısmı
    ENDWHILE

    // 9. Omleti Katla (isteğe bağlı)
    IF omlet_katlanacak THEN
        KATLA omlet
    ENDIF

    // 10. Üst Kısmı Pişir
    WHILE omlet_üstü_pişmedi DO
        BEKLE 20 saniye
        KONTROL et üst_kısmı
    ENDWHILE

    // 11. Servis Et
    AL omlet FROM tava
    SERVİS_ET omlet WITH garnitür
BİTİR
