// ANA FONKSİYON: Oynanacak en iyi kartı belirler.
// Girdi: Oyuncunun eli (elimdeki_kartlar), ortadaki kartlar (ortadaki_kartlar), lider olup olmadığı (lider_ben_miyim)
// Çıktı: Oynanacak kart nesnesi

FUNCTION PlayBestCard(elimdeki_kartlar, ortadaki_kartlar, lider_ben_miyim):

BAŞLA  
// --- DURUM 1: ELİ BAŞLATAN BENİM (LİDERİM) ---
    IF lider_ben_miyim == TRUE THEN
        // Liderken amaç, yüksek bir kartla eli kazanmaktır.
        // Strateji: En güçlü olduğun renkteki en yüksek kartı oyna. Bu, hem eli kazanma
        // şansını artırır hem de rakibin elindeki o renkten değerli kartları ortaya çıkarabilir.
        
        oynanacak_kart = NULL
        en_yuksek_deger = -1 // As'ı 14, Papaz'ı 13 vb. sayısallaştırdığımızı varsayalım.
        
        // Elimdeki her bir kartı tek tek değerlendir.
        FOR each kart IN elimdeki_kartlar:
            IF kart.deger > en_yuksek_deger THEN
                en_yuksek_deger = kart.deger
                oynanacak_kart = kart
            END IF
        END FOR
        
        // Bulunan en yüksek değerli kartı oyna.
        RETURN oynanacak_kart
    
    // --- DURUM 2: ELİ TAKİP EDEN BENİM (RAKİP BAŞLATTI) ---
    ELSE
        rakip_karti = ortadaki_kartlar[0] // Rakibin oynadığı ilk kart
        rakip_rengi = rakip_karti.renk
        
        // ADIM 2.1: Rakibin oynadığı renkten kartım var mı?
        takip_edebilecegim_kartlar = [] // O renkteki kartlarım için boş bir liste
        
        FOR each kart IN elimdeki_kartlar:
            IF kart.renk == rakip_rengi THEN
                ADD kart TO takip_edebilecegim_kartlar
            END IF
        END FOR
        
        // --- DURUM 2.A: TAKİP EDECEK KARTIM VAR ---
        IF takip_edebilecegim_kartlar IS NOT EMPTY THEN
            // Bu rengi oynamak zorundayım. Şimdi strateji devreye giriyor.
            
                       // Strateji: Rakipten daha yüksek kart varsa oyna, yoksa en düşük kartı oyna.
            rakip_deger = rakip_karti.deger
            kazanabilecek_kartlar = []
            kaybedecek_kartlar = []

            // Takip edebileceğim kartları değerlendir
            FOR each kart IN takip_edebilecegim_kartlar:
                IF kart.deger > rakip_deger THEN
                    ADD kart TO kazanabilecek_kartlar
                ELSE
                    ADD kart TO kaybedecek_kartlar
                END IF
            END FOR

            // --- DURUM 2.A.1: Rakibi geçebilecek kartım var ---
            IF kazanabilecek_kartlar IS NOT EMPTY THEN
                // En düşük kazanabilecek kartı oyna (gereksiz yere yüksek kart harcama)
                en_dusuk_kazan_kart = kazanabilecek_kartlar[0]
                FOR each kart IN kazanabilecek_kartlar:
                    IF kart.deger < en_dusuk_kazan_kart.deger THEN
                        en_dusuk_kazan_kart = kart
                    END IF
                END FOR
                RETURN en_dusuk_kazan_kart

            // --- DURUM 2.A.2: Rakibi geçemem, en düşük kartı oyna ---
            ELSE
                en_dusuk_kart = kaybedecek_kartlar[0]
                FOR each kart IN kaybedecek_kartlar:
                    IF kart.deger < en_dusuk_kart.deger THEN
                        en_dusuk_kart = kart
                    END IF
                END FOR
                RETURN en_dusuk_kart
            END IF

        // --- DURUM 2.B: Takip edecek kartım yok ---
        ELSE
            // Bu durumda istediğim kartı oynayabilirim. Strateji: En düşük değerli kartı oyna.
            en_dusuk_kart = elimdeki_kartlar[0]
            FOR each kart IN elimdeki_kartlar:
                IF kart.deger < en_dusuk_kart.deger THEN
                    en_dusuk_kart = kart
                END IF
            END FOR
            RETURN en_dusuk_kart
        END IF
    END IF
BİTİR
