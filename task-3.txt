BAŞLA   
// 1. Kullanıcıdan arama kriterlerini al
    INPUT arama_terimi
    INPUT arama_tipi // "başlık", "yazar", "ISBN", "kategori" gibi

    // 2. Arama terimini doğrula
    IF arama_terimi IS EMPTY THEN
        PRINT "Lütfen geçerli bir arama terimi giriniz."
        EXIT
    END IF

    // 3. Kitap veritabanını yükle
    kitap_listesi = VERİTABANINDAN_KİTAPLARI_GETİR()

    // 4. Arama sonuçlarını tutmak için boş liste oluştur
    bulunan_kitaplar = []

    // 5. Kitapları tek tek kontrol et
    FOR each kitap IN kitap_listesi DO

        // 5.1 Arama tipine göre eşleşme kontrolü
        IF arama_tipi == "başlık" THEN
            IF kitap.baslik CONTAINS arama_terimi THEN
                ADD kitap TO bulunan_kitaplar
            END IF

        ELSE IF arama_tipi == "yazar" THEN
            IF kitap.yazar CONTAINS arama_terimi THEN
                ADD kitap TO bulunan_kitaplar
            END IF

        ELSE IF arama_tipi == "ISBN" THEN
            IF kitap.ISBN == arama_terimi THEN
                ADD kitap TO bulunan_kitaplar
            END IF

        ELSE IF arama_tipi == "kategori" THEN
            IF kitap.kategori CONTAINS arama_terimi THEN
                ADD kitap TO bulunan_kitaplar
            END IF

        ELSE
            PRINT "Geçersiz arama tipi."
            EXIT
        END IF

    END FOR

    // 6. Arama sonuçlarını değerlendir
    IF bulunan_kitaplar IS EMPTY THEN
        PRINT "Aradığınız kriterlere uygun kitap bulunamadı."
    ELSE
        PRINT "Bulunan kitaplar:"
        FOR each kitap IN bulunan_kitaplar DO
            PRINT kitap.baslik + " - " + kitap.yazar + " - " + kitap.kategori
        END FOR
    END IF
