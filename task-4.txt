Adıl_Pasta_Paylasımı_Algorıtması

BAŞLA
    // 1. Girdi al: kişi sayısı ve pasta boyutu
    INPUT kisi_sayisi
    INPUT pasta_boyutu // örneğin gram cinsinden

    // 2. Pasta boyutunu ve kişi sayısını kontrol et
    IF kisi_sayisi <= 0 THEN
        PRINT "Geçerli bir kişi sayısı giriniz."
        EXIT
    END IF

    IF pasta_boyutu <= 0 THEN
        PRINT "Geçerli bir pasta boyutu giriniz."
        EXIT
    END IF

    // 3. Her kişiye düşen eşit pasta miktarını hesapla
    pay_miktari = pasta_boyutu / kisi_sayisi

    // 4. Paylaşım listesi oluştur
    paylasim_listesi = []

    // 5. Her kişiye sırayla pasta ver
    FOR i FROM 1 TO kisi_sayisi DO
        kişi = "Kişi_" + i
        VER kişi'ye pay_miktari gram pasta
        ADD (kişi, pay_miktari) TO paylasim_listesi
    END FOR

    // 6. Artan pasta miktarını kontrol et
    kalan_pasta = pasta_boyutu MOD kisi_sayisi

    IF kalan_pasta > 0 THEN
        PRINT "Artan pasta miktarı: " + kalan_pasta + " gram."
        // Strateji: Artanı rastgele veya sırayla kişilere dağıt
        FOR i FROM 1 TO kalan_pasta DO
            kişi_index = i MOD kisi_sayisi
            kişi = "Kişi_" + (kişi_index + 1)
            INCREMENT kişi'nin payı BY 1 gram
            GÜNCELLE paylasim_listesi[kişi]
        END FOR
    END IF

    // 7. Sonuçları yazdır
    PRINT "Adil pasta paylaşımı tamamlandı:"
    FOR each kayıt IN paylasim_listesi DO
        PRINT kayıt.kişi + " → " + kayıt.miktar + " gram"
    END FOR
